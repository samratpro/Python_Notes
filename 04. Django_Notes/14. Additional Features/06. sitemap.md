## Core urls, sitemap config
```bash
python manage.py collectstatic --noinput
```
sitemaps.py
```py
from django.contrib.sitemaps import Sitemap
from django.urls import reverse
from app1.models import *
from app2.models import *

# Sitemap for static pages
class StaticSitemap(Sitemap):
    changefreq = "weekly"
    priority = 0.8

    def items(self):
        return [
            'app:pagename1',
            'app:pagename2',
            'app:pagename3',
            'appblog:home',
        ]

    def location(self, item):
        return reverse(item)

# Sitemap for dynamic
class BlogPostSitemap(Sitemap):
    changefreq = "weekly"
    priority = 0.9

    def items(self):
        return BlogPost.objects.filter(status='Publish')

    def lastmod(self, obj):
        return obj.modified_date

    def location(self, obj):
        return f"/blog/{obj.slug}/"
```
urls.py
```py
from .sitemaps import StaticSitemap, BlogPostSitemap, CaseStudySitemap
from django.views.static import serve
from django.contrib.sitemaps.views import sitemap

sitemaps = {
    'static': StaticSitemap,
    'blog': BlogPostSitemap,
    'case_studies': CaseStudySitemap,
}

urlpatterns = [
    path('sitemap.xml', sitemap, {'sitemaps': sitemaps}, name='django.contrib.sitemaps.views.sitemap'),
    path('robots.txt', serve, {'document_root': settings.STATIC_ROOT, 'path': 'robots.txt'}),
    path('', include('app1.urls'), name='app1'),
    path('', include('app2.urls'), name='app2'),
]  
```
