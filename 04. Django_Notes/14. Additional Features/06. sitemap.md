sitemaps.py
```py
from django.contrib.sitemaps import Sitemap
from django.urls import reverse
from app1.models import *
from app2.models import *

class StaticSitemap(Sitemap):
    changefreq = "weekly"
    priority = 0.8

    def items(self):
        return [
            'app1:home',
            'app1:about',
            'app2:services',
            'app2:contact',
            'app2:blog_list',
        ]

    def location(self, item):
        return reverse(item)

class BlogPostSitemap(Sitemap):
    changefreq = "weekly"
    priority = 0.9

    def items(self):
        return BlogPost.objects.filter(status='Publish')

    def lastmod(self, obj):
        return obj.modified_date

    def location(self, obj):
        return f"/{obj.slug}/"  # Matches catch-all pattern

class CaseStudySitemap(Sitemap):
    changefreq = "monthly"
    priority = 0.7

    def items(self):
        return CaseStudy.objects.filter(status='Publish')

    def lastmod(self, obj):
        return obj.modified_date

    def location(self, obj):
        return f"/case-study/{obj.slug}/"
```
urls.py
```py
from .sitemaps import StaticSitemap, BlogPostSitemap, CaseStudySitemap
from django.views.static import serve
from django.contrib.sitemaps.views import sitemap

sitemaps = {
    'static': StaticSitemap,
    'blog': BlogPostSitemap,
    'case_studies': CaseStudySitemap,
}

urlpatterns = [
    path('admin/', admin.site.urls),
    path('sitemap.xml', sitemap, {'sitemaps': sitemaps}, name='django.contrib.sitemaps.views.sitemap'),
    path('robots.txt', serve, {'document_root': settings.STATIC_ROOT, 'path': 'robots.txt'}),
    path('', include('app1.urls', namespace='app1')),
    path('', include('app1.urls', namespace='app1')),
]
```

template/sitemap.xml
```xml
{% load l10n %}<?xml version="1.0" encoding="UTF-8"?>
<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">
{% for url in urlset %}
  <url>
    <loc>{{ url.location|escape }}</loc>
    {% if url.lastmod %}<lastmod>{{ url.lastmod|date:"Y-m-d" }}</lastmod>{% endif %}
    {% if url.changefreq %}<changefreq>{{ url.changefreq }}</changefreq>{% endif %}
    {% if url.priority %}<priority>{{ url.priority }}</priority>{% endif %}
  </url>
{% endfor %}
</urlset>
```
